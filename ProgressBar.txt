display dialog "Formula" default button 2 default answer "llvm"
set llvm to text returned of result

try
	do shell script "ls $HOME/Library/Logs/Homebrew/" & llvm & "/02.cmake 2>/dev/null"
on error
	display dialog "File not exist : $HOME/Library/Logs/Homebrew/" & llvm & "/02.cmake"
	return
end try

set progress total steps to 100
set y to 0
repeat
	do shell script "sed  '/.*make: \\*.*/!d' \\
                                            $HOME/Library/Logs/Homebrew/" & llvm & "/02.cmake 2>/dev/null"
	if not result is "" then
		display dialog llvm & " : make: Error..."
		exit repeat
	end if
	
	set x to do shell script "sed -E '/^\\[.+]/!d;s/\\[ *([0-9]+)%].+/\\1/' \\
                                            $HOME/Library/Logs/Homebrew/" & llvm & "/02.cmake 2>/dev/null"
	set str to words of x
	repeat with a in str
		set int to a + 0
		if int > y then
			set y to int
		end if
	end repeat
	set progress completed steps to y
	if y = 100 then
		exit repeat
	end if
end repeat
